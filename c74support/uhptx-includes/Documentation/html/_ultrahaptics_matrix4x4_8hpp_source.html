133  static Matrix4x4 rotate(Vector3 n, double theta)
134  {
135  double c_theta = cos(theta);
136 
137  // Return the identity matrix
138  if (c_theta >= 1.)
139  return identity();
140 
141  double s_theta = sin(theta);
142  double one_m_c = 1. - c_theta;
143 
144  Matrix4x4 m;
145 
146  m.element[ 0] = n.x * n.x * one_m_c + c_theta;
147  m.element[ 1] = n.x * n.y * one_m_c - n.z * s_theta;
148  m.element[ 2] = n.x * n.z * one_m_c + n.y * s_theta;
149  m.element[ 3] = 0.0;
150 
151  m.element[ 4] = n.y * n.x * one_m_c + n.z * s_theta;
152  m.element[ 5] = n.y * n.y * one_m_c + c_theta;
153  m.element[ 6] = n.y * n.z * one_m_c - n.x * s_theta;
154  m.element[ 7] = 0.0;
155 
156  m.element[ 8] = n.z * n.x * one_m_c - n.y * s_theta;
157  m.element[ 9] = n.z * n.y * one_m_c + n.x * s_theta;
158  m.element[10] = n.z * n.z * one_m_c + c_theta;
159  m.element[11] = 0.0;
160 
161  m.element[12] = 0.0;
162  m.element[13] = 0.0;
163  m.element[14] = 0.0;
164  m.element[15] = 1.0;
165 
166  return m;
167  }

176  static Matrix4x4 rotateOntoVector(Vector3 p, Vector3 q)
177  {
178  Vector3 cross_pq = p.cross(q);
179  double dot_pq = p.dot(q);
180 
181  if (dot_pq <= -1.)
182  return rotate(Vector3(1.0f, 0.0f, 0.0f), M_PI);
183 
184  double rcp_1p_dot_pq = 1. / (1. + dot_pq);
185 
186  Matrix4x4 m;
187 
188  m.element[ 0] = cross_pq.x * cross_pq.x * rcp_1p_dot_pq + dot_pq;
189  m.element[ 1] = cross_pq.x * cross_pq.y * rcp_1p_dot_pq - cross_pq.z;
190  m.element[ 2] = cross_pq.x * cross_pq.z * rcp_1p_dot_pq + cross_pq.y;
191  m.element[ 3] = 0.0;
192 
193  m.element[ 4] = cross_pq.y * cross_pq.x * rcp_1p_dot_pq + cross_pq.z;
194  m.element[ 5] = cross_pq.y * cross_pq.y * rcp_1p_dot_pq + dot_pq;
195  m.element[ 6] = cross_pq.y * cross_pq.z * rcp_1p_dot_pq - cross_pq.x;
196  m.element[ 7] = 0.0;
197 
198  m.element[ 8] = cross_pq.z * cross_pq.x * rcp_1p_dot_pq - cross_pq.y;
199  m.element[ 9] = cross_pq.z * cross_pq.y * rcp_1p_dot_pq + cross_pq.x;
200  m.element[10] = cross_pq.z * cross_pq.z * rcp_1p_dot_pq + dot_pq;
